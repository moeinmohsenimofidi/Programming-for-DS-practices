# write your code here
X, y = make_blobs(n_samples=200, centers=2, random_state=0)

model = svm.SVC(kernel="linear", C=1000)
model.fit(X, y)

plt.scatter(X[:, 0], X[:, 1], c=y, s=20, cmap="winter")

ax = plt.gca()
xlim = ax.get_xlim()
ylim = ax.get_ylim()


xx = np.linspace(xlim[0], xlim[1], 200)
yy = np.linspace(ylim[0], ylim[1], 200)
YY, XX = np.meshgrid(yy, xx)
xy = np.vstack([XX.ravel(), YY.ravel()]).T
Z = model.decision_function(xy).reshape(XX.shape)

ax.contour(
    XX, YY, Z, colors="k", levels=[-1, 0, 1], alpha=0.5, linestyles=["--", "-", "--"]
)

plt.show()

def r2Scores(C):
    X, y = make_circles( n_samples=200,  factor=0.6,noise=0.30, random_state =0)
    
    model = SVC(kernel = 'rbf', C = C)
    model.fit(X, y);


    return r2_score(y, model.predict(X));
    
score = []
for i in range(1, 200):
    score.append(r2Scores(i))
    
plt.plot(range(1, 200),score,color='blue')
plt.title('Change in R2 Scores')
plt.xlabel('C')
plt.ylabel('Score of R2 from previous excercise')
plt.show()
